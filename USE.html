# 快速上手

# 安装

npm install vue-drag-man --save

# 使用

在main.js中引入并注册组件

// main.js
import vueDragMan from 'vue-drag-man'
Vue.use(vueDragMan)

然后在组件中就可以直接使用<Drag>标签和<Drop>标签

<!-- 被<Drag>标签包裹的元素会成为"可拖拽"元素 -->
<Drag>
    <div class="drag">drag</div>
</Drag>

<!-- 被<Drop>标签包裹的元素会成为"可放置"元素 -->
<Drop>
    <div class="drop"></div>
</Drop>

# 示例

# 基础示例

将需要被拖拽的元素使用<Drag>包裹起来, 将需要被放置的区域使用<Drop>包裹起来即可

<template>
  <div>
    <Drag>
      <!-- 被拖拽的目标 -->
      <div class="dragged">target</div>
    </Drag>
    <Drop>
      <!-- 目标放置区域 -->
      <div class="drop-area"></div>
    </Drop>
  </div>
</template>

# 事件示例

拖放过程包括6个事件, 回调参数为原生事件对象

<template>
  <div id="app">
    <Drag @drag="drag" @dragend="dragend">
      <div class="dragged">target</div>
    </Drag>
    <Drop @dragenter="dragenter" @dragover="dragover" @dragleave="dragover" @drop="drop">
      <div class="drop-area"></div>
    </Drop>
  </div>
</template>

<script>
export default {
  name: 'app',
  methods: {
    drag(e) {
      console.log('拖放过程中', e)
    },
    dragend(e) {
      console.log('拖放结束', e)
    },
    dragenter(e) {
      console.log('源对象 开始进入过程对象范围内', e)
    },
    dragover(e) {
      console.log('源对象 在过程对象范围内移动', e)
    },
    dragleave(e) {
      console.log('源对象 离开过程对象的范围', e)
    },
    drop(el, e) {
      console.log('源对象 被拖放到目标对象内', el, e)
    }
  }
}
</script>

# Events

源对象(<Drag>)事件

事件名称 说明 回调参数
dragstart 源对象开始拖放 事件对象Event
drag 源对象拖放过程中 事件对象Event
dragend 源对象拖放结束 事件对象Event

目标对象(<Drop>)事件

事件名称 说明 回调参数
dragenter 源对象开始进入目标对象范围内 事件对象Event
dragover 源对象在目标对象范围内移动 事件对象Event
dragleave 源对象离开目标对象的范围 事件对象Event
drop 源对象被拖放到目标对象内 拖拽节点Node | 事件对象Event
